    # Solution 1:
    fuel_req_fn = function(module_req){
      req = sum(floor(module_req/3)-2)
      total = req
      while(req > 0){
        req = sum(floor(req/3)-2)
        if(req < 1){break}
        total = c(total, req)
      }
      return(total)
    }

    map(scan("01-input"), fuel_req_fn) %>% unlist() %>% sum()
